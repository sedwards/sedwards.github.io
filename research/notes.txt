  <!-- Include any other meta tags or SEO information here -->
</head>
<body>
  <!-- Minimalist theme structure -->
  <header>
    <h1>Welcome to My Minimalist Site</h1>
    <nav>
      <ul>
        <li><a href="index.html">Home</a></li>
        <li><a href="about.html">About</a></li>
        <li><a href="contact.html">Contact</a></li>
      </ul>
    </nav>
  </header>

  <!-- Section where the Markdown content will be injected -->
  <main id="content">
    <!-- JavaScript will dynamically load the markdown-generated HTML here -->
  </main>

  <footer>
    <p>Â© 2024 Minimalist Theme</p>
  </footer>

  <!-- Link to the theme's JavaScript (if needed) -->
  <script src="path/to/your/theme.js"></script>

  <!-- Script to dynamically load and inject rendered Markdown HTML -->
  <script>
    // Fetch and insert the markdown-rendered HTML into the content area
    fetch('path/to/your/markdown-page.html')  // Change the path to the correct HTML file
      .then(response => response.text())
      .then(html => {
        document.getElementById('content').innerHTML = html;
      })
      .catch(error => {
        console.error('Error loading content:', error);
      });
  </script>
</body>
</html>
